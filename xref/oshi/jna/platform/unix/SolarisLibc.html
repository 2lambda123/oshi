40      int UTX_USERSIZE = 32;
41      int UTX_LINESIZE = 32;
42      int UTX_IDSIZE = 4;
43      int UTX_HOSTSIZE = 257;
44  
45      int PRCLSZ = 8;
46      int PRFNSZ = 16;
47      int PRLNSZ = 32;
48      int PRARGSZ = 80;

50      /**
51       * Connection info
52       */
53      @FieldOrder({ "ut_user", "ut_id", "ut_line", "ut_pid", "ut_type", "ut_tv", "ut_session", "ut_syslen", "ut_host" })
54      class SolarisUtmpx extends Structure {
55          public byte[] ut_user = new byte[UTX_USERSIZE]; // user login name
56          public byte[] ut_id = new byte[UTX_IDSIZE]; // etc/inittab id (usually line #)
57          public byte[] ut_line = new byte[UTX_LINESIZE]; // device name
58          public int ut_pid; // process id
59          public short ut_type; // type of entry
60          public Timeval ut_tv; // time entry was made
61          public int ut_session; // session ID, used for windowing
62          public short ut_syslen; // significant length of ut_host including terminating null
63          public byte[] ut_host = new byte[UTX_HOSTSIZE]; // host name
64      }
65  
66      /**
67       * Part of utmpx structure
68       */
69      @FieldOrder({ "e_termination", "e_exit" })
70      class Exit_status extends Structure {
71          public short e_termination; // Process termination status
72          public short e_exit; // Process exit status
73      }

75      /**
76       * 32/64-bit timeval required for utmpx structure
77       */
78      @FieldOrder({ "tv_sec", "tv_usec" })
79      class Timeval extends Structure {
80          public NativeLong tv_sec; // seconds
81          public NativeLong tv_usec; // microseconds
82      }

85       * Reads a line from the current file position in the utmp file. It returns a

93      SolarisUtmpx getutxent();

95      /**
96       * Structure for psinfo file
97       */
98      @FieldOrder({ "pr_flag", "pr_nlwp", "pr_pid", "pr_ppid", "pr_pgid", "pr_sid", "pr_uid", "pr_euid", "pr_gid",
99              "pr_egid", "pr_addr", "pr_size", "pr_rssize", "pr_rssizepriv", "pr_ttydev", "pr_pctcpu", "pr_pctmem",
100             "pr_start", "pr_time", "pr_ctime", "pr_fname", "pr_psargs", "pr_wstat", "pr_argc", "pr_argv", "pr_envp",
101             "pr_dmodel", "pr_pad2", "pr_taskid", "pr_projid", "pr_nzomb", "pr_poolid", "pr_zoneid", "pr_contract",
102             "pr_filler", "pr_lwp" })
103     class SolarisPsInfo extends Structure {
104         public int pr_flag; // process flags (DEPRECATED; do not use)
105         public int pr_nlwp; // number of active lwps in the process
106         public int pr_pid; // unique process id
107         public int pr_ppid; // process id of parent
108         public int pr_pgid; // pid of process group leader
109         public int pr_sid; // session id
110         public int pr_uid; // real user id
111         public int pr_euid; // effective user id
112         public int pr_gid; // real group id
113         public int pr_egid; // effective group id
114         public Pointer pr_addr; // address of process
115         public size_t pr_size; // size of process image in Kbytes
116         public size_t pr_rssize; // resident set size in Kbytes
117         public size_t pr_rssizepriv; // resident set size of private mappings
118         public NativeLong pr_ttydev; // controlling tty device (or PRNODEV)
119         // The following percent numbers are 16-bit binary
120         // fractions [0 .. 1] with the binary point to the
121         // right of the high-order bit (1.0 == 0x8000)
122         public short pr_pctcpu; // % of recent cpu time used by all lwps
123         public short pr_pctmem; // % of system memory used by process
124         public Timestruc pr_start; // process start time, from the epoch
125         public Timestruc pr_time; // usr+sys cpu time for this process
126         public Timestruc pr_ctime; // usr+sys cpu time for reaped children
127         public byte[] pr_fname = new byte[PRFNSZ]; // name of exec'ed file
128         public byte[] pr_psargs = new byte[PRARGSZ]; // initial characters of arg list
129         public int pr_wstat; // if zombie, the wait() status
130         public int pr_argc; // initial argument count
131         public Pointer pr_argv; // address of initial argument vector
132         public Pointer pr_envp; // address of initial environment vector
133         public byte pr_dmodel; // data model of the process
134         public byte[] pr_pad2 = new byte[3];
135         public int pr_taskid; // task id
136         public int pr_projid; // project id
137         public int pr_nzomb; // number of zombie lwps in the process
138         public int pr_poolid; // pool id
139         public int pr_zoneid; // zone id
140         public int pr_contract; // process contract id
141         public int[] pr_filler = new int[1]; // reserved for future use
142         public SolarisLwpsInfo pr_lwp; // information for representative lwp
143 
144         public SolarisPsInfo() {
145             super();
146         }
147 
148         public SolarisPsInfo(byte[] bytes) {
149             super();
150             // Truncate bytes and pad with 0 if necessary
151             byte[] structBytes = new byte[size()];
152             System.arraycopy(bytes, 0, structBytes, 0, structBytes.length);
153             // Write bytes to native
154             this.getPointer().write(0, structBytes, 0, structBytes.length);
155             // Read bytes to struct
156             read();
157         }
158     }

160     /**
161      * Nested Structure for psinfo file
162      */
163     @FieldOrder({ "pr_flag", "pr_lwpid", "pr_addr", "pr_wchan", "pr_stype", "pr_state", "pr_sname", "pr_nice",
164             "pr_syscall", "pr_oldpri", "pr_cpu", "pr_pri", "pr_pctcpu", "pr_pad", "pr_start", "pr_time", "pr_clname",
165             "pr_oldname", "pr_onpro", "pr_bindpro", "pr_bindpset", "pr_lgrp", "pr_last_onproc", "pr_name" })
166     class SolarisLwpsInfo extends Structure {
167         public int pr_flag; // lwp flags (DEPRECATED; do not use)
168         public int pr_lwpid; // lwp id
169         public Pointer pr_addr; // DEPRECATED was internal address of lwp
170         public Pointer pr_wchan; // DEPRECATED was wait addr for sleeping lwp
171         public byte pr_stype; // synchronization event type
172         public byte pr_state; // numeric lwp state
173         public byte pr_sname; // printable character for pr_state
174         public byte pr_nice; // nice for cpu usage
175         public short pr_syscall; // system call number (if in syscall)
176         public byte pr_oldpri; // pre-SVR4, low value is high priority
177         public byte pr_cpu; // pre-SVR4, cpu usage for scheduling
178         public int pr_pri; // priority, high value = high priority
179         // The following percent numbers are 16-bit binary
180         // fractions [0 .. 1] with the binary point to the
181         // right of the high-order bit (1.0 == 0x8000)
182         public short pr_pctcpu; // % of recent cpu time used by this lwp
183         public short pr_pad;
184         public Timestruc pr_start; // lwp start time, from the epoch
185         public Timestruc pr_time; // cpu time for this lwp
186         public byte[] pr_clname = new byte[PRCLSZ]; // scheduling class name
187         public byte[] pr_oldname = new byte[PRFNSZ]; // binary compatibility -- unused
188         public int pr_onpro; // processor which last ran this lwp
189         public int pr_bindpro; // processor to which lwp is bound
190         public int pr_bindpset; // processor set to which lwp is bound
191         public int pr_lgrp; // home lgroup
192         public long pr_last_onproc; // Timestamp of when thread last ran on a processor
193         public byte[] pr_name = new byte[PRLNSZ]; // name of system lwp
194

196             super();
197         }

204             // Write bytes to native
205             this.getPointer().write(0, structBytes, 0, structBytes.length);
206             // Read bytes to struct
207             read();
208         }
209     }

211     /**
212      * Structure for usage file
213      */
214     @FieldOrder({ "pr_lwpid", "pr_count", "pr_tstamp", "pr_create", "pr_term", "pr_rtime", "pr_utime", "pr_stime",
215             "pr_ttime", "pr_tftime", "pr_dftime", "pr_kftime", "pr_ltime", "pr_slptime", "pr_wtime", "pr_stoptime",
216             "filltime", "pr_minf", "pr_majf", "pr_nswap", "pr_inblk", "pr_oublk", "pr_msnd", "pr_mrcv", "pr_sigs",
217             "pr_vctx", "pr_ictx", "pr_sysc", "pr_ioch", "filler" })
218     class SolarisPrUsage extends Structure {
219         public int pr_lwpid; // lwp id. 0: process or defunct
220         public int pr_count; // number of contributing lwps
221         public Timestruc pr_tstamp; // current time stamp
222         public Timestruc pr_create; // process/lwp creation time stamp
223         public Timestruc pr_term; // process/lwp termination time stamp
224         public Timestruc pr_rtime; // total lwp real (elapsed) time
225         public Timestruc pr_utime; // user level cpu time
226         public Timestruc pr_stime; // system call cpu time
227         public Timestruc pr_ttime; // other system trap cpu time
228         public Timestruc pr_tftime; // text page fault sleep time
229         public Timestruc pr_dftime; // data page fault sleep time
230         public Timestruc pr_kftime; // kernel page fault sleep time
231         public Timestruc pr_ltime; // user lock wait sleep time
232         public Timestruc pr_slptime; // all other sleep time
233         public Timestruc pr_wtime; // wait-cpu (latency) time
234         public Timestruc pr_stoptime; // stopped time
235         public Timestruc[] filltime = new Timestruc[6]; // filler for future expansion
236         public NativeLong pr_minf; // minor page faults
237         public NativeLong pr_majf; // major page faults
238         public NativeLong pr_nswap; // swaps
239         public NativeLong pr_inblk; // input blocks
240         public NativeLong pr_oublk; // output blocks
241         public NativeLong pr_msnd; // messages sent
242         public NativeLong pr_mrcv; // messages received
243         public NativeLong pr_sigs; // signals received
244         public NativeLong pr_vctx; // voluntary context switches
245         public NativeLong pr_ictx; // involuntary context switches
246         public NativeLong pr_sysc; // system calls
247         public NativeLong pr_ioch; // chars read and written
248         public NativeLong[] filler = new NativeLong[10]; // filler for future expansion
