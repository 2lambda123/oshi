54      /**
55       * Test read file.
56       */
57      @Test
58      void testReadFile() {
59          // Write to a temp file
60          Path multilineFile = null;
61          try {
62              multilineFile = Files.createTempFile("oshitest.multiline", null);
63              String s = "Line 1\nLine 2\nThe third line\nLine 4\nLine 5\n";
64              Files.write(multilineFile, s.getBytes(StandardCharsets.UTF_8));
65          } catch (IOException e) {
66              fail("IO Exception creating or writing to temporary multiline file.");
67          }
68  
69          // Try the new temp file
70          List<String> tempFileStrings = FileUtil.readFile(multilineFile.toString());
71          assertThat("Temp file line one mismatch", tempFileStrings.get(0), is("Line 1"));
72          List<String> matchingLines = tempFileStrings.stream().filter(s -> s.startsWith("Line "))
73                  .collect(Collectors.toList());
74          assertThat("Matching lines mismatch", matchingLines.size(), is(4));
75  
76          // Delete the temp file
77          try {
78              Files.deleteIfExists(multilineFile);
79          } catch (IOException e) {
80              fail("IO Exception deleting temporary multiline file.");
81          }
82  
83          // Try file not found on deleted file
84          assertThat("Deleted file should return empty", FileUtil.readFile(multilineFile.toString()), is(empty()));
85      }

87      /**
88       * Test get*FromFile
89       */
90      @Test
91      void testGetFromFile() {
92          // Write to temp file
93          Path integerFile = null;
94          try {
95              integerFile = Files.createTempFile("oshitest.int", null);
96              Files.write(integerFile, "123\n".getBytes(StandardCharsets.UTF_8));
97          } catch (IOException e) {
98              fail("IO Exception creating or writing to temporary integer file.");
99          }
100         assertThat("unsigned long from int", FileUtil.getUnsignedLongFromFile(integerFile.toString()), is(123L));
101         assertThat("long from int", FileUtil.getLongFromFile(integerFile.toString()), is(123L));
102         assertThat("int from int", FileUtil.getIntFromFile(integerFile.toString()), is(123));
103         assertThat("string from int", FileUtil.getStringFromFile(integerFile.toString()), is("123"));
104 
105         // Delete the temp file
106         try {
107             Files.deleteIfExists(integerFile);
108         } catch (IOException e) {
109             fail("IO Exception deleting temporary integer file.");
110         }
111 
112         // Write to temp file
113         Path stringFile = null;
114         try {
115             stringFile = Files.createTempFile("oshitest.str", null);
116             Files.write(stringFile, "foo bar\n".getBytes(StandardCharsets.UTF_8));
117         } catch (IOException e) {
118             fail("IO Exception creating or writing to temporary string file.");
119         }
120 
121         assertThat("unsigned long from string", FileUtil.getUnsignedLongFromFile(stringFile.toString()), is(0L));
122         assertThat("long from string", FileUtil.getLongFromFile(stringFile.toString()), is(0L));
123         assertThat("int from string", FileUtil.getIntFromFile(stringFile.toString()), is(0));
124         assertThat("string from string", FileUtil.getStringFromFile(stringFile.toString()), is("foo bar"));
125         // Delete the temp file
126         try {
127             Files.deleteIfExists(stringFile);
128         } catch (IOException e) {
129             fail("IO Exception deleting temporary string file.");
130         }
131 
132         // Try file not found on deleted file
133         assertThat("unsigned long from invalid", FileUtil.getUnsignedLongFromFile(stringFile.toString()), is(0L));
134         assertThat("long from invalid", FileUtil.getLongFromFile(stringFile.toString()), is(0L));
135         assertThat("int from invalid", FileUtil.getIntFromFile(stringFile.toString()), is(0));
136         assertThat("string from invalid ", FileUtil.getStringFromFile(stringFile.toString()), is(emptyString()));
137     }

139     @Test
140     void testReadProcIo() {
141         Map<String, String> expected = new LinkedHashMap<>();
142         expected.put("rchar", "124788352");
143         expected.put("wchar", "124802481");
144         expected.put("syscr", "135");
145         expected.put("syscw", "1547");
146         expected.put("read_bytes", "40304640");
147         expected.put("write_bytes", "124780544");
148         expected.put("cancelled_write_bytes", "42");
149         // Write this to a temp file
150         Path procIoFile = null;
151         try {
152             procIoFile = Files.createTempFile("oshitest.procio", null);
153             for (Entry<String, String> e : expected.entrySet()) {
154                 String s = e.getKey() + ": " + e.getValue() + "\n";
155                 Files.write(procIoFile, s.getBytes(StandardCharsets.UTF_8), StandardOpenOption.APPEND);
156             }
157         } catch (IOException e) {
158             fail("IO Exception creating or writing to temporary procIo file.");
159         }
160         // Read into map
161         Map<String, String> actual = FileUtil.getKeyValueMapFromFile(procIoFile.toString(), ":");
162         assertThat("procio size", actual, is(aMapWithSize(expected.size())));
163         for (Entry<String, String> entry : expected.entrySet()) {
164             assertThat("procio entry", actual, hasEntry(entry.getKey(), entry.getValue()));
165         }
166 
167         // Cleanup
168         try {
169             Files.deleteIfExists(procIoFile);
170         } catch (IOException e) {
171             fail("IO Exception deleting temporary procIo file.");
172         }
173 
174         // Test deleted file
175         actual = FileUtil.getKeyValueMapFromFile(procIoFile.toString(), ":");
176         assertThat("procio size", actual, anEmptyMap());
177     }
