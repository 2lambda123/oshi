24  package oshi.jna.platform.unix;
25  
26  import com.sun.jna.Native; // NOSONAR squid:S1191
27  import com.sun.jna.Structure;
28  import com.sun.jna.Structure.FieldOrder;
29  
30  /**
31   * C library. This class should be considered non-API as it may be removed
32   * if/when its code is incorporated into the JNA project.
33   */
34  public interface AixLibc extends CLibrary {
35  
36      AixLibc INSTANCE = Native.load("c", AixLibc.class);
37  
38      int PRCLSZ = 8;
39      int PRFNSZ = 16;
40      int PRARGSZ = 80;
41  
42      @FieldOrder({ "pr_flag", "pr_flag2", "pr_nlwp", "pr__pad1", "pr_uid", "pr_euid", "pr_gid", "pr_egid", "pr_pid",
43              "pr_ppid", "pr_pgid", "pr_sid", "pr_ttydev", "pr_addr", "pr_size", "pr_rssize", "pr_start", "pr_time",
44              "pr_cid", "pr__pad2", "pr_argc", "pr_argv", "pr_envp", "pr_fname", "pr_psargs", "pr__pad", "pr_lwp" })
45      class AixPsInfo extends Structure {
46          public int pr_flag; // process flags from proc struct p_flag
47          public int pr_flag2; // process flags from proc struct p_flag2
48          public int pr_nlwp; // number of threads in process
49          public int pr__pad1; // reserved for future use
50          public long pr_uid; // real user id
51          public long pr_euid; // effective user id
52          public long pr_gid; // real group id
53          public long pr_egid; // effective group id
54          public long pr_pid; // unique process id
55          public long pr_ppid; // process id of parent
56          public long pr_pgid; // pid of process group leader
57          public long pr_sid; // session id
58          public long pr_ttydev; // controlling tty device
59          public long pr_addr; // internal address of proc struct
60          public long pr_size; // size of process image in KB (1024) units
61          public long pr_rssize; // resident set size in KB (1024) units
62          public Timestruc pr_start; // process start time, time since epoch
63          public Timestruc pr_time; // usr+sys cpu time for this process
64          public short pr_cid; // corral id
65          public short pr__pad2; // reserved for future use
66          public int pr_argc; // initial argument count
67          public long pr_argv; // address of initial argument vector in user process
68          public long pr_envp; // address of initial environment vector in user process
69          public byte[] pr_fname = new byte[PRFNSZ]; // last component of exec()ed pathname
70          public byte[] pr_psargs = new byte[PRARGSZ]; // initial characters of arg list
71          public long[] pr__pad = new long[8]; // reserved for future use
72          public AIXLwpsInfo pr_lwp; // "representative" thread info
73  
74          public AixPsInfo() {
75              super();
76          }
77  
78          public AixPsInfo(byte[] bytes) {
79              super();
80              // Truncate bytes and pad with 0 if necessary
81              byte[] structBytes = new byte[size()];
82              System.arraycopy(bytes, 0, structBytes, 0, structBytes.length);
83              // Write bytes to native
84              this.getPointer().write(0, structBytes, 0, structBytes.length);
85              // Read bytes to struct
86              read();
87          }
88      }
89  
90      @FieldOrder({ "pr_lwpid", "pr_addr", "pr_wchan", "pr_flag", "pr_wtype", "pr_state", "pr_sname", "pr_nice", "pr_pri",
91              "pr_policy", "pr_clname", "pr_onpro", "pr_bindpro" })
92      class AIXLwpsInfo extends Structure {
93          public long pr_lwpid; // thread id
94          public long pr_addr; // internal address of thread
95          public long pr_wchan; // wait addr for sleeping thread
96          public int pr_flag; // thread flags
97          public byte pr_wtype; // type of thread wait
98          public byte pr_state; // numeric scheduling state
99          public byte pr_sname; // printable character representing pr_state
100         public byte pr_nice; // nice for cpu usage
101         public int pr_pri; // priority, high value = high priority
102         public int pr_policy; // scheduling policy
103         public byte[] pr_clname = new byte[PRCLSZ]; // printable character representing pr_policy
104         public int pr_onpro; // processor on which thread last ran
105         public int pr_bindpro; // processor to which thread is bound
106 
107         public AIXLwpsInfo() {
108             super();
109         }
110 
111         public AIXLwpsInfo(byte[] bytes) {
112             super();
113             // Truncate bytes and pad with 0 if necessary
114             byte[] structBytes = new byte[size()];
115             System.arraycopy(bytes, 0, structBytes, 0, structBytes.length);
116             // Write bytes to native
117             this.getPointer().write(0, structBytes, 0, structBytes.length);
118             // Read bytes to struct
119             read();
120         }
121     }
122 
123     /**
124      * 64-bit timestruc required for psinfo structure
125      */
126     @FieldOrder({ "tv_sec", "tv_nsec", "pad" })
127     class Timestruc extends Structure {
128         public long tv_sec; // seconds
129         public int tv_nsec; // nanoseconds
130         public int pad; // nanoseconds
131     }
132 }
