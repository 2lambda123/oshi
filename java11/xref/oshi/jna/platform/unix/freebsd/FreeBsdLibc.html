41      int UTX_USERSIZE = 32;
42      int UTX_LINESIZE = 16;
43      int UTX_IDSIZE = 8;
44      int UTX_HOSTSIZE = 128;

46      /**
47       * Connection info
48       */
49      @FieldOrder({ "ut_type", "ut_tv", "ut_id", "ut_pid", "ut_user", "ut_line", "ut_host", "ut_spare" })
50      class FreeBsdUtmpx extends Structure {
51          public short ut_type; // type of entry
52          public Timeval ut_tv; // time entry was made
53          public byte[] ut_id = new byte[UTX_IDSIZE]; // etc/inittab id (usually line #)
54          public int ut_pid; // process id
55          public byte[] ut_user = new byte[UTX_USERSIZE]; // user login name
56          public byte[] ut_line = new byte[UTX_LINESIZE]; // device name
57          public byte[] ut_host = new byte[UTX_HOSTSIZE]; // host name
58          public byte[] ut_spare = new byte[64];
59      }

67      int INT_SIZE = Native.getNativeSize(int.class);

102     /**
103      * The sysctl() function retrieves system information and allows processes with
104      * appropriate privileges to set system information. The information available
105      * from sysctl() consists of integers, strings, and tables.
106      *
107      * The state is described using a "Management Information Base" (MIB) style
108      * name, listed in name, which is a namelen length array of integers.
109      *
110      * The information is copied into the buffer specified by oldp. The size of the
111      * buffer is given by the location specified by oldlenp before the call, and
112      * that location gives the amount of data copied after a successful call and
113      * after a call that returns with the error code ENOMEM. If the amount of data
114      * available is greater than the size of the buffer supplied, the call supplies
115      * as much data as fits in the buffer provided and returns with the error code
116      * ENOMEM. If the old value is not desired, oldp and oldlenp should be set to
117      * NULL.
118      *
119      * The size of the available data can be determined by calling sysctl() with the
120      * NULL argument for oldp. The size of the available data will be returned in
121      * the location pointed to by oldlenp. For some operations, the amount of space
122      * may change often. For these operations, the system attempts to round up so
123      * that the returned size is large enough for a call to return the data shortly
124      * thereafter.
125      *
126      * To set a new value, newp is set to point to a buffer of length newlen from
127      * which the requested value is to be taken. If a new value is not to be set,
128      * newp should be set to NULL and newlen set to 0.
129      *
130      * @param name
131      *            MIB array of integers
132      * @param namelen
133      *            length of the MIB array
134      * @param oldp
135      *            Information retrieved
136      * @param oldlenp
137      *            Size of information retrieved
138      * @param newp
139      *            Information to be written
140      * @param newlen
141      *            Size of information to be written
142      * @return 0 on success; sets errno on failure
143      */
144     int sysctl(int[] name, int namelen, Pointer oldp, IntByReference oldlenp, Pointer newp, int newlen);
145 
146     /**
147      * The sysctlbyname() function accepts an ASCII representation of the name and
148      * internally looks up the integer name vector. Apart from that, it behaves the
149      * same as the standard sysctl() function.
150      *
151      * @param name
152      *            ASCII representation of the MIB name
153      * @param oldp
154      *            Information retrieved
155      * @param oldlenp
156      *            Size of information retrieved
157      * @param newp
158      *            Information to be written
159      * @param newlen
160      *            Size of information to be written
161      * @return 0 on success; sets errno on failure
162      */
163     int sysctlbyname(String name, Pointer oldp, IntByReference oldlenp, Pointer newp, int newlen);
164 
165     /**
166      * The sysctlnametomib() function accepts an ASCII representation of the name,
167      * looks up the integer name vector, and returns the numeric representation in
168      * the mib array pointed to by mibp. The number of elements in the mib array is
169      * given by the location specified by sizep before the call, and that location
170      * gives the number of entries copied after a successful call. The resulting mib
171      * and size may be used in subsequent sysctl() calls to get the data associated
172      * with the requested ASCII name. This interface is intended for use by
173      * applications that want to repeatedly request the same variable (the sysctl()
174      * function runs in about a third the time as the same request made via the
175      * sysctlbyname() function).
176      *
177      * The number of elements in the mib array can be determined by calling
178      * sysctlnametomib() with the NULL argument for mibp.
179      *
180      * The sysctlnametomib() function is also useful for fetching mib prefixes. If
181      * size on input is greater than the number of elements written, the array still
182      * contains the additional elements which may be written programmatically.
183      *
184      * @param name
185      *            ASCII representation of the name
186      * @param mibp
187      *            Integer array containing the corresponding name vector.
188      * @param size
189      *            On input, number of elements in the returned array; on output, the
190      *            number of entries copied.
191      * @return 0 on success; sets errno on failure
192      */
193     int sysctlnametomib(String name, Pointer mibp, IntByReference size);
194 
195     /**
196      * Reads a line from the current file position in the utmp file. It returns a
197      * pointer to a structure containing the fields of the line.
198      * <p>
199      * Not thread safe
200      *
201      * @return a {@link FreeBsdUtmpx} on success, and NULL on failure (which
202      *         includes the "record not found" case)
203      */
204     FreeBsdUtmpx getutxent();
205 }
