169     @Test
170     void testByteArrayToFloat() {
171         byte[] temp = { (byte) 0x12, (byte) 0x34, (byte) 0x56, (byte) 0x78, (byte) 0x9a };
172         float f = (temp[0] << 22 | temp[1] << 14 | temp[2] << 6 | temp[3] >>> 2) + (float) (temp[3] & 0x3) / 0x4;

174         f = 0x12345 + (float) 0x6 / 0x10;
175         assertEquals(f, ParseUtil.byteArrayToFloat(temp, 3, 4), Float.MIN_VALUE);
176         f = 0x123 + (float) 0x4 / 0x10;
177         assertEquals(f, ParseUtil.byteArrayToFloat(temp, 2, 4), Float.MIN_VALUE);

347      * Test parse string to array
348      */
349     @Test
350     void testParseStringToLongArray() {
351         int[] indices = { 1, 3 };
352         long now = System.currentTimeMillis();
353 
354         String foo = String.format("The numbers are %d %d %d %d", 123, 456, 789, now);
355         int count = ParseUtil.countStringToLongArray(foo, ' ');
356         assertThat("countStringToLongArray should return 4 for \"" + foo + "\"", count, is(4));
357         long[] result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
358         assertThat("result[0] should be 456 using parseStringToLongArray on \"" + foo + "\"", result[0], is(456L));
359         assertThat("result[1] should be " + now + " using parseStringToLongArray on \"" + foo + "\"", result[1],
360                 is(now));
361 
362         foo = String.format("The numbers are %d %d %d %d %s", 123, 456, 789, now,
363                 "709af748-5f8e-41b3-b73a-b440ef4406c8");
364         count = ParseUtil.countStringToLongArray(foo, ' ');
365         assertThat("countStringToLongArray should return 4 for \"" + foo + "\"", count, is(4));
366         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
367         assertThat("result[0] should be 456 using parseStringToLongArray on \"" + foo + "\"", result[0], is(456L));
368         assertThat("result[1] should be " + now + " using parseStringToLongArray on \"" + foo + "\"", result[1],
369                 is(now));
370 
371         foo = String.format("The numbers are %d -%d %d +%d", 123, 456, 789, now);
372         count = ParseUtil.countStringToLongArray(foo, ' ');
373         assertThat("countStringToLongArray should return 4 for \"" + foo + "\"", count, is(4));
374         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
375         assertThat("result[0] should be -4456 using parseStringToLongArray on \"" + foo + "\"", result[0], is(-456L));
376         assertThat("result[1] index should be 456 using parseStringToLongArray on \"" + foo + "\"", result[1], is(now));
377 
378         foo = String.format("NOLOG: Invalid character %d %s %d %d", 123, "4v6", 789, now);
379         count = ParseUtil.countStringToLongArray(foo, ' ');
380         assertThat("countStringToLongArray should return 2 for \"" + foo + "\"", count, is(2));
381         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
382         assertThat("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", result[1], is(0L));
383 
384         foo = String.format("Exceeds max long %d %d %d 1%d", 123, 456, 789, Long.MAX_VALUE);
385         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
386         assertThat("result[1] index should be " + Long.MAX_VALUE
387                 + " (Long.MAX_VALUE) using parseStringToLongArray on \"" + foo + "\"", result[1], is(Long.MAX_VALUE));
388 
389         foo = String.format("NOLOG: String too short %d %d %d %d", 123, 456, 789, now);

391         assertThat("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", result[1], is(0L));
392

394         result = ParseUtil.parseStringToLongArray(foo, indices, 2, ' ');
395         assertThat("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", result[1], is(0L));

397         foo = String.format("%d %d %d %d", 123, 456, 789, now);
398         count = ParseUtil.countStringToLongArray(foo, ' ');
