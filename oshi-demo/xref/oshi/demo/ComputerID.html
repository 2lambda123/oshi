5   package oshi.demo;

7   import oshi.SystemInfo;
8   import oshi.hardware.CentralProcessor;
9   import oshi.hardware.ComputerSystem;
10  import oshi.hardware.HardwareAbstractionLayer;
11  import oshi.software.os.OperatingSystem;
12  import oshi.util.Constants;
13  
14  /**
15   * Attempts to create a unique computer identifier. Note that serial numbers won't work on Linux without user
16   * cooperation.
17   */
18  public class ComputerID {
19      /**
20       * <p>
21       * main.
22       * </p>
23       *
24       * @param args an array of {@link java.lang.String} objects.
25       */
26      public static void main(String[] args) {
27          String unknownHash = String.format("%08x", Constants.UNKNOWN.hashCode());
28  
29          System.out.println("Here's a unique (?) id for your computer.");
30          System.out.println(getComputerIdentifier());
31          System.out.println("If any field is " + unknownHash
32                  + " then I couldn't find a serial number or uuid, and running as sudo might change this.");
33      }
34  
35      /**
36       * Generates a Computer Identifier, which may be part of a strategy to construct a licence key. (The identifier may
37       * not be unique as in one case hashcode could be same for multiple values, and the result may differ based on
38       * whether the program is running with sudo/root permission.) The identifier string is based upon the processor
39       * serial number, vendor, processor identifier, and total processor count.
40       *
41       * @return A string containing four hyphen-delimited fields representing the processor; the first 3 are 32-bit
42       *         hexadecimal values and the last one is an integer value.
43       */
44      public static String getComputerIdentifier() {
45          SystemInfo systemInfo = new SystemInfo();
46          OperatingSystem operatingSystem = systemInfo.getOperatingSystem();
47          HardwareAbstractionLayer hardwareAbstractionLayer = systemInfo.getHardware();
48          CentralProcessor centralProcessor = hardwareAbstractionLayer.getProcessor();
49          ComputerSystem computerSystem = hardwareAbstractionLayer.getComputerSystem();
50  
51          String vendor = operatingSystem.getManufacturer();
52          String processorSerialNumber = computerSystem.getSerialNumber();
53          String uuid = computerSystem.getHardwareUUID();
54          String processorIdentifier = centralProcessor.getProcessorIdentifier().getIdentifier();
55          int processors = centralProcessor.getLogicalProcessorCount();
56  
57          String delimiter = "-";
58  
59          return String.format("%08x", vendor.hashCode()) + delimiter
60                  + String.format("%08x", processorSerialNumber.hashCode()) + delimiter
61                  + String.format("%08x", uuid.hashCode()) + delimiter
62                  + String.format("%08x", processorIdentifier.hashCode()) + delimiter + processors;
63      }
64  }
