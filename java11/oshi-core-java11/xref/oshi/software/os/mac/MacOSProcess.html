132     private String queryCommandLine() {
133         // Get command line via sysctl
134         int[] mib = new int[3];
135         mib[0] = 1; // CTL_KERN
136         mib[1] = 49; // KERN_PROCARGS2
137         mib[2] = getProcessID();
138         // Allocate memory for arguments
139         int argmax = SysctlUtil.sysctl("kern.argmax", 0);
140         Pointer procargs = new Memory(argmax);
141         IntByReference size = new IntByReference(argmax);
142         // Fetch arguments
143         if (0 != SystemB.INSTANCE.sysctl(mib, mib.length, procargs, size, null, 0)) {
144             // Beginning in macOS 11, this call has become unreliable. Use ps as a backup.
145             String cmdLine = ExecutingCommand.getFirstAnswer("ps -o command= -p " + getProcessID());
146             if (!cmdLine.isEmpty()) {
147                 return cmdLine;
148             }
149             LOG.warn(
150                     "Failed syctl call for process arguments (kern.procargs2), process {} may not exist. Error code: {}",
151                     getProcessID(), Native.getLastError());
152             return "";
153         }
154         // Procargs contains an int representing total # of args, followed by a
155         // null-terminated execpath string and then the arguments, each
156         // null-terminated (possible multiple consecutive nulls),
157         // The execpath string is also the first arg.
158         int nargs = procargs.getInt(0);
159         // Sanity check
160         if (nargs < 0 || nargs > 1024) {
161             LOG.error("Nonsensical number of process arguments for pid {}: {}", getProcessID(), nargs);
162             return "";
163         }
164         List<String> args = new ArrayList<>(nargs);
165         // Skip first int (containing value of nargs)
166         long offset = SystemB.INT_SIZE;
167         // Skip exec_command
168         offset += procargs.getString(offset).length();
169         // Iterate character by character using offset
170         // Build each arg and add to list
171         while (nargs-- > 0 && offset < size.getValue()) {
172             // Advance through additional nulls
173             while (procargs.getByte(offset) == 0) {
174                 if (++offset >= size.getValue()) {
175                     break;
176                 }
177             }
178             // Grab a string. This should go until the null terminator
179             String arg = procargs.getString(offset);
180             args.add(arg);
181             // Advance offset to next null
182             offset += arg.length();
183         }
184         // Return args null-delimited
185         return String.join("\0", args);
186     }

228     @Override
229     public List<OSThread> getThreadDetails() {
230         long now = System.currentTimeMillis();
231         List<OSThread> details = new ArrayList<>();
232         List<ThreadStats> stats = ThreadInfo.queryTaskThreads(getProcessID());
233         for (ThreadStats stat : stats) {
234             // For long running threads the start time calculation can overestimate
235             long start = now - stat.getUpTime();
236             if (start < this.getStartTime()) {
237                 start = this.getStartTime();
238             }
239             details.add(new MacOSThread(getProcessID(), stat.getThreadId(), stat.getState(), stat.getSystemTime(),
240                     stat.getUserTime(), start, now - start, stat.getPriority()));
241         }
242         return details;
243     }

300     @Override
301     public long getAffinityMask() {
302         // macOS doesn't do affinity. Return a bitmask of the current processors.
303         int logicalProcessorCount = SysctlUtil.sysctl("hw.logicalcpu", 1);
304         return logicalProcessorCount < 64 ? (1L << logicalProcessorCount) - 1 : -1L;
305     }

317     @Override
318     public boolean updateAttributes() {
319         long now = System.currentTimeMillis();
320         ProcTaskAllInfo taskAllInfo = new ProcTaskAllInfo();
321         if (0 > SystemB.INSTANCE.proc_pidinfo(getProcessID(), SystemB.PROC_PIDTASKALLINFO, 0, taskAllInfo,
322                 taskAllInfo.size()) || taskAllInfo.ptinfo.pti_threadnum < 1) {
323             this.state = INVALID;
324             return false;
325         }
326         Pointer buf = new Memory(SystemB.PROC_PIDPATHINFO_MAXSIZE);
327         if (0 < SystemB.INSTANCE.proc_pidpath(getProcessID(), buf, SystemB.PROC_PIDPATHINFO_MAXSIZE)) {
328             this.path = buf.getString(0).trim();
329             // Overwrite name with last part of path
330             String[] pathSplit = this.path.split("/");
331             if (pathSplit.length > 0) {
332                 this.name = pathSplit[pathSplit.length - 1];
333             }
334         }
335         if (this.name.isEmpty()) {
336             // pbi_comm contains first 16 characters of name
337             this.name = Native.toString(taskAllInfo.pbsd.pbi_comm, StandardCharsets.UTF_8);
338         }
339 
340         switch (taskAllInfo.pbsd.pbi_status) {
341         case SSLEEP:
342             this.state = SLEEPING;
343             break;
344         case SWAIT:
345             this.state = WAITING;
346             break;
347         case SRUN:
348             this.state = RUNNING;
349             break;
350         case SIDL:
351             this.state = NEW;
352             break;
353         case SZOMB:
354             this.state = ZOMBIE;
355             break;
356         case SSTOP:
357             this.state = STOPPED;
358             break;
359         default:
360             this.state = OTHER;
361             break;
362         }
363         this.parentProcessID = taskAllInfo.pbsd.pbi_ppid;
364         this.userID = Integer.toString(taskAllInfo.pbsd.pbi_uid);
365         Passwd pwuid = SystemB.INSTANCE.getpwuid(taskAllInfo.pbsd.pbi_uid);
366         if (pwuid != null) {
367             this.user = pwuid.pw_name;
368         }
369         this.groupID = Integer.toString(taskAllInfo.pbsd.pbi_gid);
370         Group grgid = SystemB.INSTANCE.getgrgid(taskAllInfo.pbsd.pbi_gid);
371         if (grgid != null) {
372             this.group = grgid.gr_name;
373         }
374         this.threadCount = taskAllInfo.ptinfo.pti_threadnum;
375         this.priority = taskAllInfo.ptinfo.pti_priority;
376         this.virtualSize = taskAllInfo.ptinfo.pti_virtual_size;
377         this.residentSetSize = taskAllInfo.ptinfo.pti_resident_size;
378         this.kernelTime = taskAllInfo.ptinfo.pti_total_system / 1_000_000L;
379         this.userTime = taskAllInfo.ptinfo.pti_total_user / 1_000_000L;
380         this.startTime = taskAllInfo.pbsd.pbi_start_tvsec * 1000L + taskAllInfo.pbsd.pbi_start_tvusec / 1000L;
381         this.upTime = now - this.startTime;
382         this.openFiles = taskAllInfo.pbsd.pbi_nfiles;
383         this.bitness = (taskAllInfo.pbsd.pbi_flags & P_LP64) == 0 ? 32 : 64;
384         this.majorFaults = taskAllInfo.ptinfo.pti_pageins;
385         // testing using getrusage confirms pti_faults includes both major and minor
386         this.minorFaults = taskAllInfo.ptinfo.pti_faults - taskAllInfo.ptinfo.pti_pageins;
387         if (this.minorVersion >= 9) {
388             RUsageInfoV2 rUsageInfoV2 = new RUsageInfoV2();
389             if (0 == SystemB.INSTANCE.proc_pid_rusage(getProcessID(), SystemB.RUSAGE_INFO_V2, rUsageInfoV2)) {
390                 this.bytesRead = rUsageInfoV2.ri_diskio_bytesread;
391                 this.bytesWritten = rUsageInfoV2.ri_diskio_byteswritten;
392             }
393         }
394         VnodePathInfo vpi = new VnodePathInfo();
395         if (0 < SystemB.INSTANCE.proc_pidinfo(getProcessID(), SystemB.PROC_PIDVNODEPATHINFO, 0, vpi, vpi.size())) {
396             this.currentWorkingDirectory = Native.toString(vpi.pvi_cdir.vip_path, StandardCharsets.US_ASCII);
397         }
398         return true;
399     }
