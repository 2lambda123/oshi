54  /**
55   * Provides access to kstat information on Solaris
56   */
57  @ThreadSafe
58  public final class KstatUtil {
59  
60      private static final Logger LOG = LoggerFactory.getLogger(KstatUtil.class);
61  
62      private static final Lock CHAIN = new ReentrantLock();
63      // Only one thread may access the chain at any time, so we wrap this object in
64      // the KstatChain class locked until the lock is released on auto-close.
65      @GuardedBy("CHAIN")
66      private static KstatCtl kstatCtl = null;
67  
68      private KstatUtil() {
69      }
70  
71      /**
72       * A copy of the Kstat chain, encapsulating a {@code kstat_ctl_t} object. Only
73       * one thread may actively use this object at any time.
74       * <p>
75       * The chain is created once calling {@link LibKstat#kstat_open} and then this
76       * object is instantiated using the {@link KstatUtil#openChain} method.
77       * Instantiating this object updates the chain using
78       * {@link LibKstat#kstat_chain_update}. The control object should be closed with
79       * {@link #close}, which releases the lock and allows another instance to be
80       * instantiated.
81       */
82      public static final class KstatChain implements AutoCloseable {
83  
84          private final KstatCtl localCtlRef;
85  
86          private KstatChain(KstatCtl ctl) {
87              this.localCtlRef = ctl;
88              update();
89          }
90  
91          /**
92           * Convenience method for {@link LibKstat#kstat_read} which gets data from the
93           * kernel for the kstat pointed to by {@code ksp}. {@code ksp.ks_data} is
94           * automatically allocated (or reallocated) to be large enough to hold all of
95           * the data. {@code ksp.ks_ndata} is set to the number of data fields,
96           * {@code ksp.ks_data_size} is set to the total size of the data, and
97           * ksp.ks_snaptime is set to the high-resolution time at which the data snapshot
98           * was taken.
99           *
100          * @param ksp
101          *            The kstat from which to retrieve data
102          * @return {@code true} if successful; {@code false} otherwise
103          */
104         @GuardedBy("CHAIN")
105         public boolean read(Kstat ksp) {
106             int retry = 0;
107             while (0 > LibKstat.INSTANCE.kstat_read(localCtlRef, ksp, null)) {
108                 if (LibKstat.EAGAIN != Native.getLastError() || 5 <= ++retry) {
109                     if (LOG.isDebugEnabled()) {
110                         LOG.debug("Failed to read kstat {}:{}:{}",
111                                 Native.toString(ksp.ks_module, StandardCharsets.US_ASCII), ksp.ks_instance,
112                                 Native.toString(ksp.ks_name, StandardCharsets.US_ASCII));
113                     }
114                     return false;
115                 }
116                 Util.sleep(8 << retry);
117             }
118             return true;
119         }
120 
121         /**
122          * Convenience method for {@link LibKstat#kstat_lookup}. Traverses the kstat
123          * chain, searching for a kstat with the same {@code module}, {@code instance},
124          * and {@code name} fields; this triplet uniquely identifies a kstat. If
125          * {@code module} is {@code null}, {@code instance} is -1, or {@code name} is
126          * {@code null}, then those fields will be ignored in the search.
127          *
128          * @param module
129          *            The module, or null to ignore
130          * @param instance
131          *            The instance, or -1 to ignore
132          * @param name
133          *            The name, or null to ignore
134          * @return The first match of the requested Kstat structure if found, or
135          *         {@code null}
136          */
137         @GuardedBy("CHAIN")
138         public Kstat lookup(String module, int instance, String name) {
139             return LibKstat.INSTANCE.kstat_lookup(localCtlRef, module, instance, name);
140         }
141 
142         /**
143          * Convenience method for {@link LibKstat#kstat_lookup}. Traverses the kstat
144          * chain, searching for all kstats with the same {@code module},
145          * {@code instance}, and {@code name} fields; this triplet uniquely identifies a
146          * kstat. If {@code module} is {@code null}, {@code instance} is -1, or
147          * {@code name} is {@code null}, then those fields will be ignored in the
148          * search.
149          *
150          * @param module
151          *            The module, or null to ignore
152          * @param instance
153          *            The instance, or -1 to ignore
154          * @param name
155          *            The name, or null to ignore
156          * @return All matches of the requested Kstat structure if found, or an empty
157          *         list otherwise
158          */
159         @GuardedBy("CHAIN")
160         public List<Kstat> lookupAll(String module, int instance, String name) {
161             List<Kstat> kstats = new ArrayList<>();
162             for (Kstat ksp = LibKstat.INSTANCE.kstat_lookup(localCtlRef, module, instance, name); ksp != null; ksp = ksp
163                     .next()) {
164                 if ((module == null || module.equals(Native.toString(ksp.ks_module, StandardCharsets.US_ASCII)))
165                         && (instance < 0 || instance == ksp.ks_instance)
166                         && (name == null || name.equals(Native.toString(ksp.ks_name, StandardCharsets.US_ASCII)))) {
167                     kstats.add(ksp);
168                 }
169             }
170             return kstats;
171         }
172 
173         /**
174          * Convenience method for {@link LibKstat#kstat_chain_update}. Brings this kstat
175          * header chain in sync with that of the kernel.
176          * <p>
177          * This function compares the kernel's current kstat chain ID(KCID), which is
178          * incremented every time the kstat chain changes, to this object's KCID.
179          *
180          * @return the new KCID if the kstat chain has changed, 0 if it hasn't, or -1 on
181          *         failure.
182          */
183         @GuardedBy("CHAIN")
184         public int update() {
185             return LibKstat.INSTANCE.kstat_chain_update(localCtlRef);
186         }
187 
188         /**
189          * Release the lock on the chain.
190          */
191         @Override
192         public void close() {
193             CHAIN.unlock();
194         }
195     }
196 
197     /**
198      * Lock the Kstat chain for use by this object until it's closed.
199      *
200      * @return A locked copy of the chain. It should be unlocked/released when you
201      *         are done with it with {@link KstatChain#close()}.
202      */
203     public static synchronized KstatChain openChain() {
204         CHAIN.lock();
205         if (kstatCtl == null) {
206             kstatCtl = LibKstat.INSTANCE.kstat_open();
207         }
208         return new KstatChain(kstatCtl);
209     }

211     /**
212      * Convenience method for {@link LibKstat#kstat_data_lookup} with String return
213      * values. Searches the kstat's data section for the record with the specified
214      * name. This operation is valid only for kstat types which have named data
215      * records. Currently, only the KSTAT_TYPE_NAMED and KSTAT_TYPE_TIMER kstats
216      * have named data records.
217      *
218      * @param ksp
219      *            The kstat to search
220      * @param name
221      *            The key for the name-value pair, or name of the timer as
222      *            applicable
223      * @return The value as a String.
224      */
225     public static String dataLookupString(Kstat ksp, String name) {
226         if (ksp.ks_type != LibKstat.KSTAT_TYPE_NAMED && ksp.ks_type != LibKstat.KSTAT_TYPE_TIMER) {
227             throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
228         }
229         Pointer p = LibKstat.INSTANCE.kstat_data_lookup(ksp, name);
230         if (p == null) {
231             LOG.debug("Failed to lookup kstat value for key {}", name);
232             return "";
233         }
234         KstatNamed data = new KstatNamed(p);
235         switch (data.data_type) {
236         case LibKstat.KSTAT_DATA_CHAR:
237             return Native.toString(data.value.charc, StandardCharsets.UTF_8);
238         case LibKstat.KSTAT_DATA_INT32:
239             return Integer.toString(data.value.i32);
240         case LibKstat.KSTAT_DATA_UINT32:
241             return FormatUtil.toUnsignedString(data.value.ui32);
242         case LibKstat.KSTAT_DATA_INT64:
243             return Long.toString(data.value.i64);
244         case LibKstat.KSTAT_DATA_UINT64:
245             return FormatUtil.toUnsignedString(data.value.ui64);
246         case LibKstat.KSTAT_DATA_STRING:
247             return data.value.str.addr.getString(0);
248         default:
249             LOG.error("Unimplemented kstat data type {}", data.data_type);
250             return "";
251         }
252     }

254     /**
255      * Convenience method for {@link LibKstat#kstat_data_lookup} with numeric return
256      * values. Searches the kstat's data section for the record with the specified
257      * name. This operation is valid only for kstat types which have named data
258      * records. Currently, only the KSTAT_TYPE_NAMED and KSTAT_TYPE_TIMER kstats
259      * have named data records.
260      *
261      * @param ksp
262      *            The kstat to search
263      * @param name
264      *            The key for the name-value pair, or name of the timer as
265      *            applicable
266      * @return The value as a long. If the data type is a character or string type,
267      *         returns 0 and logs an error.
268      */
269     public static long dataLookupLong(Kstat ksp, String name) {
270         if (ksp.ks_type != LibKstat.KSTAT_TYPE_NAMED && ksp.ks_type != LibKstat.KSTAT_TYPE_TIMER) {
271             throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
272         }
273         Pointer p = LibKstat.INSTANCE.kstat_data_lookup(ksp, name);
274         if (p == null) {
275             if (LOG.isDebugEnabled()) {
276                 LOG.debug("Failed lo lookup kstat value on {}:{}:{} for key {}",
277                         Native.toString(ksp.ks_module, StandardCharsets.US_ASCII), ksp.ks_instance,
278                         Native.toString(ksp.ks_name, StandardCharsets.US_ASCII), name);
279             }
280             return 0L;
281         }
282         KstatNamed data = new KstatNamed(p);
283         switch (data.data_type) {
284         case LibKstat.KSTAT_DATA_INT32:
285             return data.value.i32;
286         case LibKstat.KSTAT_DATA_UINT32:
287             return FormatUtil.getUnsignedInt(data.value.ui32);
288         case LibKstat.KSTAT_DATA_INT64:
289             return data.value.i64;
290         case LibKstat.KSTAT_DATA_UINT64:
291             return data.value.ui64;
292         default:
293             LOG.error("Unimplemented or non-numeric kstat data type {}", data.data_type);
294             return 0L;
295         }
296     }

298     /**
299      * Query Kstat2 with a single map
300      *
301      * @param mapStr
302      *            The map to query
303      * @param names
304      *            Names of data to query
305      * @return An object array with the data corresponding to the names
306      */
307     public static Object[] queryKstat2(String mapStr, String... names) {
308         if (!SolarisOperatingSystem.HAS_KSTAT2) {
309             throw new UnsupportedOperationException(
310                     "Kstat2 requires Solaris 11.4+. Use SolarisOperatingSystem#HAS_KSTAT2 to test this.");
311         }
312         Object[] result = new Object[names.length];
313         Kstat2MatcherList matchers = new Kstat2MatcherList();
314         KstatUtil.CHAIN.lock();
315         try {
316             matchers.addMatcher(Kstat2.KSTAT2_M_STRING, mapStr);
317             Kstat2Handle handle = new Kstat2Handle();
318             try {
319                 Kstat2Map map = handle.lookupMap(mapStr);
320                 for (int i = 0; i < names.length; i++) {
321                     result[i] = map.getValue(names[i]);
322                 }
323             } finally {
324                 handle.close();
325             }
326         } catch (Kstat2StatusException e) {
327             LOG.debug("Failed to get stats on {} for names {}: {}", mapStr, Arrays.toString(names), e.getMessage());
328         } finally {
329             KstatUtil.CHAIN.unlock();
330             matchers.free();
331         }
332         return result;
333     }

335     /**
336      * Query Kstat2 iterating over maps using a wildcard indicating a 0-indexed
337      * list, such as a cpu.
338      *
339      * @param beforeStr
340      *            The part of the string before the wildcard
341      * @param afterStr
342      *            The part of the string after the wildcard
343      * @param names
344      *            Names of data to query
345      * @return A list of object arrays with the data corresponding to the names
346      */
347     public static List<Object[]> queryKstat2List(String beforeStr, String afterStr, String... names) {
348         if (!SolarisOperatingSystem.HAS_KSTAT2) {
349             throw new UnsupportedOperationException(
350                     "Kstat2 requires Solaris 11.4+. Use SolarisOperatingSystem#HAS_KSTAT2 to test this.");
351         }
352         List<Object[]> results = new ArrayList<>();
353         int s = 0;
354         Kstat2MatcherList matchers = new Kstat2MatcherList();
355         KstatUtil.CHAIN.lock();
356         try {
357             matchers.addMatcher(Kstat2.KSTAT2_M_GLOB, beforeStr + "*" + afterStr);
358             Kstat2Handle handle = new Kstat2Handle();
359             try {
360                 for (s = 0; s < Integer.MAX_VALUE; s++) {
361                     Object[] result = new Object[names.length];
362                     Kstat2Map map = handle.lookupMap(beforeStr + s + afterStr);
363                     for (int i = 0; i < names.length; i++) {
364                         result[i] = map.getValue(names[i]);
365                     }
366                     results.add(result);
367                 }
368             } finally {
369                 handle.close();
370             }
371         } catch (Kstat2StatusException e) {
372             // Expected to end iteration
373             LOG.debug("Failed to get stats on {}{}{} for names {}: {}", beforeStr, s, afterStr, Arrays.toString(names),
374                     e.getMessage());
375         } finally {
376             KstatUtil.CHAIN.unlock();
377             matchers.free();
378         }
379         return results;
380     }
