50      /**
51       * Gets Battery Information
52       *
53       * @return An array of PowerSource objects representing batteries, etc.
54       */
55      public static List<PowerSource> getPowerSources() {
56          if (!HAS_UDEV) {
57              LOG.warn("Power Source information requires libudev, which is not present.");
58              return Collections.emptyList();
59          }
60          String psName;
61          String psDeviceName;
62          double psRemainingCapacityPercent = -1d;
63          double psTimeRemainingEstimated = -1d; // -1 = unknown, -2 = unlimited
64          double psTimeRemainingInstant = -1d;
65          double psPowerUsageRate = 0d;
66          double psVoltage = -1d;
67          double psAmperage = 0d;
68          boolean psPowerOnLine = false;
69          boolean psCharging = false;
70          boolean psDischarging = false;
71          CapacityUnits psCapacityUnits = CapacityUnits.RELATIVE;
72          int psCurrentCapacity = -1;
73          int psMaxCapacity = -1;
74          int psDesignCapacity = -1;
75          int psCycleCount = -1;
76          String psChemistry;
77          LocalDate psManufactureDate = null;
78          String psManufacturer;
79          String psSerialNumber;
80          double psTemperature = 0d;
81  
82          List<PowerSource> psList = new ArrayList<>();
83          UdevContext udev = Udev.INSTANCE.udev_new();
84          try {
85              UdevEnumerate enumerate = udev.enumerateNew();
86              try {
87                  enumerate.addMatchSubsystem("power_supply");
88                  enumerate.scanDevices();
89                  for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
90                      String syspath = entry.getName();
91                      String name = syspath.substring(syspath.lastIndexOf(File.separatorChar) + 1);
92                      if (!name.startsWith("ADP") && !name.startsWith("AC")) {
93                          UdevDevice device = udev.deviceNewFromSyspath(syspath);
94                          if (device != null) {
95                              try {
96                                  if (ParseUtil.parseIntOrDefault(device.getPropertyValue("POWER_SUPPLY_PRESENT"), 1) > 0
97                                          && ParseUtil.parseIntOrDefault(device.getPropertyValue("POWER_SUPPLY_ONLINE"),
98                                                  1) > 0) {
99                                      psName = getOrDefault(device, "POWER_SUPPLY_NAME", name);
100                                     String status = device.getPropertyValue("POWER_SUPPLY_STATUS");
101                                     psCharging = "Charging".equals(status);
102                                     psDischarging = "Discharging".equals(status);
103                                     psRemainingCapacityPercent = ParseUtil.parseIntOrDefault(
104                                             device.getPropertyValue("POWER_SUPPLY_CAPACITY"), -100) / 100d;
105 
106                                     // Debian/Ubuntu provides Energy. Fedora/RHEL provides Charge.
107                                     psCurrentCapacity = ParseUtil
108                                             .parseIntOrDefault(device.getPropertyValue("POWER_SUPPLY_ENERGY_NOW"), -1);
109                                     if (psCurrentCapacity < 0) {
110                                         psCurrentCapacity = ParseUtil.parseIntOrDefault(
111                                                 device.getPropertyValue("POWER_SUPPLY_CHARGE_NOW"), -1);
112                                     }
113                                     psMaxCapacity = ParseUtil
114                                             .parseIntOrDefault(device.getPropertyValue("POWER_SUPPLY_ENERGY_FULL"), 1);
115                                     if (psMaxCapacity < 0) {
116                                         psMaxCapacity = ParseUtil.parseIntOrDefault(
117                                                 device.getPropertyValue("POWER_SUPPLY_CHARGE_FULL"), 1);
118                                     }
119                                     psDesignCapacity = ParseUtil.parseIntOrDefault(
120                                             device.getPropertyValue("POWER_SUPPLY_ENERGY_FULL_DESIGN"), 1);
121                                     if (psDesignCapacity < 0) {
122                                         psDesignCapacity = ParseUtil.parseIntOrDefault(
123                                                 device.getPropertyValue("POWER_SUPPLY_CHARGE_FULL_DESIGN"), 1);
124                                     }
125 
126                                     // Debian/Ubuntu provides Voltage and Power.
127                                     // Fedora/RHEL provides Voltage and Current.
128                                     psVoltage = ParseUtil
129                                             .parseIntOrDefault(device.getPropertyValue("POWER_SUPPLY_VOLTAGE_NOW"), -1);
130                                     // From Physics we know P = IV so I = P/V
131                                     if (psVoltage > 0) {
132                                         String power = device.getPropertyValue("POWER_SUPPLY_POWER_NOW");
133                                         String current = device.getPropertyValue("POWER_SUPPLY_CURRENT_NOW");
134                                         if (power == null) {
135                                             psAmperage = ParseUtil.parseIntOrDefault(current, 0);
136                                             psPowerUsageRate = psAmperage * psVoltage;
137                                         } else if (current == null) {
138                                             psPowerUsageRate = ParseUtil.parseIntOrDefault(power, 0);
139                                             psAmperage = psPowerUsageRate / psVoltage;
140                                         } else {
141                                             psAmperage = ParseUtil.parseIntOrDefault(current, 0);
142                                             psPowerUsageRate = ParseUtil.parseIntOrDefault(power, 0);
143                                         }
144                                     }
145 
146                                     psCycleCount = ParseUtil
147                                             .parseIntOrDefault(device.getPropertyValue("POWER_SUPPLY_CYCLE_COUNT"), -1);
148                                     psChemistry = getOrDefault(device, "POWER_SUPPLY_TECHNOLOGY", Constants.UNKNOWN);
149                                     psDeviceName = getOrDefault(device, "POWER_SUPPLY_MODEL_NAME", Constants.UNKNOWN);
150                                     psManufacturer = getOrDefault(device, "POWER_SUPPLY_MANUFACTURER",
151                                             Constants.UNKNOWN);
152                                     psSerialNumber = getOrDefault(device, "POWER_SUPPLY_SERIAL_NUMBER",
153                                             Constants.UNKNOWN);
154 
155                                     psList.add(new LinuxPowerSource(psName, psDeviceName, psRemainingCapacityPercent,
156                                             psTimeRemainingEstimated, psTimeRemainingInstant, psPowerUsageRate,
157                                             psVoltage, psAmperage, psPowerOnLine, psCharging, psDischarging,
158                                             psCapacityUnits, psCurrentCapacity, psMaxCapacity, psDesignCapacity,
159                                             psCycleCount, psChemistry, psManufactureDate, psManufacturer,
160                                             psSerialNumber, psTemperature));
161                                 }
162                             } finally {
163                                 device.unref();
164                             }
165                         }
166                     }
167                 }
168             } finally {
169                 enumerate.unref();
170             }
171         } finally {
172             udev.unref();
173         }
174         return psList;
175     }
176 
177     private static String getOrDefault(UdevDevice device, String property, String def) {
178         String value = device.getPropertyValue(property);
179         return value == null ? def : value;
180     }
