85      @Override
86      protected ProcessorIdentifier queryProcessorId() {
87          String cpuVendor = "";
88          String cpuName = "";
89          String cpuIdentifier = "";
90          String cpuFamily = "";
91          String cpuModel = "";
92          String cpuStepping = "";
93          long cpuVendorFreq = 0L;
94          String processorID;
95          boolean cpu64bit = false;
96  
97          final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
98          String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
99          if (processorIds.length > 0) {
100             String cpuRegistryPath = cpuRegistryRoot + processorIds[0];
101             cpuVendor = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
102                     "VendorIdentifier");
103             cpuName = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
104                     "ProcessorNameString");
105             cpuIdentifier = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
106                     "Identifier");
107             try {
108                 cpuVendorFreq = Advapi32Util.registryGetIntValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "~MHz")
109                         * 1_000_000L;
110             } catch (Win32Exception e) {
111                 // Leave as 0, parse the identifier as backup
112             }
113         }
114         if (!cpuIdentifier.isEmpty()) {
115             cpuFamily = parseIdentifier(cpuIdentifier, "Family");
116             cpuModel = parseIdentifier(cpuIdentifier, "Model");
117             cpuStepping = parseIdentifier(cpuIdentifier, "Stepping");
118         }
119         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
120         Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
121         int processorArchitecture = sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue();
122         if (processorArchitecture == 9 // PROCESSOR_ARCHITECTURE_AMD64
123                 || processorArchitecture == 12 // PROCESSOR_ARCHITECTURE_ARM64
124                 || processorArchitecture == 6) { // PROCESSOR_ARCHITECTURE_IA64
125             cpu64bit = true;
126         }
127         WmiResult<ProcessorIdProperty> processorId = Win32Processor.queryProcessorId();
128         if (processorId.getResultCount() > 0) {
129             processorID = WmiUtil.getString(processorId, ProcessorIdProperty.PROCESSORID, 0);
130         } else {
131             processorID = createProcessorID(cpuStepping, cpuModel, cpuFamily,
132                     cpu64bit ? new String[] { "ia64" } : new String[0]);
133         }
134         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
135                 cpuVendorFreq);
136     }

138     /**
139      * Parses identifier string
140      *
141      * @param identifier
142      *            the full identifier string
143      * @param key
144      *            the key to retrieve
145      * @return the string following id
146      */
147     private static String parseIdentifier(String identifier, String key) {
148         String[] idSplit = ParseUtil.whitespaces.split(identifier);
149         boolean found = false;
150         for (String s : idSplit) {
151             // If key string found, return next value
152             if (found) {
153                 return s;
154             }
155             found = s.equals(key);
156         }
157         // If key string not found, return empty string
158         return "";
159     }

161     @Override
162     protected Pair<List<LogicalProcessor>, List<PhysicalProcessor>> initProcessorCounts() {
163         if (VersionHelpers.IsWindows7OrGreater()) {
164             Pair<List<LogicalProcessor>, List<PhysicalProcessor>> procs = LogicalProcessorInformation
165                     .getLogicalProcessorInformationEx();
166             // Save numaNode,Processor lookup for future PerfCounter instance lookup
167             // The processor number is based on the Processor Group, so we keep a separate
168             // index by NUMA node.
169             int curNode = -1;
170             int procNum = 0;
171             // 0-indexed list of all lps for array lookup
172             int lp = 0;
173             this.numaNodeProcToLogicalProcMap = new HashMap<>();
174             for (LogicalProcessor logProc : procs.getA()) {
175                 int node = logProc.getNumaNode();
176                 // This list is grouped by NUMA node so a change in node will reset this counter
177                 if (node != curNode) {
178                     curNode = node;
179                     procNum = 0;
180                 }
181                 numaNodeProcToLogicalProcMap.put(String.format("%d,%d", logProc.getNumaNode(), procNum++), lp++);
182             }
183             return procs;
184         } else {
185             return LogicalProcessorInformation.getLogicalProcessorInformation();
186         }
187     }

189     @Override
190     public long[] querySystemCpuLoadTicks() {
191         // To get load in processor group scenario, we need perfmon counters, but the
192         // _Total instance is an average rather than total (scaled) number of ticks
193         // which matches GetSystemTimes() results. We can just query the per-processor
194         // ticks and add them up. Calling the get() method gains the benefit of
195         // synchronizing this output with the memoized result of per-processor ticks as
196         // well.
197         long[] ticks = new long[TickType.values().length];
198         // Sum processor ticks
199         long[][] procTicks = getProcessorCpuLoadTicks();
200         for (int i = 0; i < ticks.length; i++) {
201             for (long[] procTick : procTicks) {
202                 ticks[i] += procTick[i];
203             }
204         }
205         return ticks;
206     }

208     @Override
209     public long[] queryCurrentFreq() {
210         if (VersionHelpers.IsWindows7OrGreater()) {
211             Pair<List<String>, Map<ProcessorFrequencyProperty, List<Long>>> instanceValuePair = ProcessorInformation
212                     .queryFrequencyCounters();
213             List<String> instances = instanceValuePair.getA();
214             Map<ProcessorFrequencyProperty, List<Long>> valueMap = instanceValuePair.getB();
215             List<Long> percentMaxList = valueMap.get(ProcessorFrequencyProperty.PERCENTOFMAXIMUMFREQUENCY);
216             if (!instances.isEmpty()) {
217                 long maxFreq = this.getMaxFreq();
218                 long[] freqs = new long[getLogicalProcessorCount()];
219                 for (int p = 0; p < instances.size(); p++) {
220                     int cpu = instances.get(p).contains(",")
221                             ? numaNodeProcToLogicalProcMap.getOrDefault(instances.get(p), 0)
222                             : ParseUtil.parseIntOrDefault(instances.get(p), 0);
223                     if (cpu >= getLogicalProcessorCount()) {
224                         continue;
225                     }
226                     freqs[cpu] = percentMaxList.get(cpu) * maxFreq / 100L;
227                 }
228                 return freqs;
229             }
230         }
231         // If <Win7 or anything failed in PDH/WMI, use the native call
232         return queryNTPower(2); // Current is field index 2
233     }

235     @Override
236     public long queryMaxFreq() {
237         long[] freqs = queryNTPower(1); // Max is field index 1
238         return Arrays.stream(freqs).max().orElse(-1L);
239     }

241     /**
242      * Call CallNTPowerInformation for Processor information and return an array of
243      * the specified index
244      *
245      * @param fieldIndex
246      *            The field, in order as defined in the
247      *            {@link PowrProf#PROCESSOR_INFORMATION} structure.
248      * @return The array of values.
249      */
250     private long[] queryNTPower(int fieldIndex) {
251         ProcessorPowerInformation ppi = new ProcessorPowerInformation();
252         long[] freqs = new long[getLogicalProcessorCount()];
253         int bufferSize = ppi.size() * freqs.length;
254         Memory mem = new Memory(bufferSize);
255         if (0 != PowrProf.INSTANCE.CallNtPowerInformation(POWER_INFORMATION_LEVEL.ProcessorInformation, null, 0, mem,
256                 bufferSize)) {
257             LOG.error("Unable to get Processor Information");
258             Arrays.fill(freqs, -1L);
259             return freqs;
260         }
261         for (int i = 0; i < freqs.length; i++) {
262             ppi = new ProcessorPowerInformation(mem.share(i * (long) ppi.size()));
263             if (fieldIndex == 1) { // Max
264                 freqs[i] = ppi.maxMhz * 1_000_000L;
265             } else if (fieldIndex == 2) { // Current
266                 freqs[i] = ppi.currentMhz * 1_000_000L;
267             } else {
268                 freqs[i] = -1L;
269             }
270         }
271         return freqs;
272     }

274     @Override
275     public double[] getSystemLoadAverage(int nelem) {
276         if (nelem < 1 || nelem > 3) {
277             throw new IllegalArgumentException("Must include from one to three elements.");
278         }
279         double[] average = new double[nelem];
280         // Windows doesn't have load average
281         for (int i = 0; i < average.length; i++) {
282             average[i] = -1;
283         }
284         return average;
285     }

287     @Override
288     public long[][] queryProcessorCpuLoadTicks() {
289         // These are used in all cases
290         List<String> instances;
291         List<Long> systemList;
292         List<Long> userList;
293         List<Long> irqList;
294         List<Long> softIrqList;
295         List<Long> idleList;
296         // These are only used with USE_CPU_UTILITY
297         List<Long> baseList = null;
298         List<Long> systemUtility = null;
299         List<Long> processorUtility = null;
300         List<Long> processorUtilityBase = null;
301         List<Long> initSystemList = null;
302         List<Long> initUserList = null;
303         List<Long> initBase = null;
304         List<Long> initSystemUtility = null;
305         List<Long> initProcessorUtility = null;
306         List<Long> initProcessorUtilityBase = null;
307         if (ProcessorInformation.USE_CPU_UTILITY) {
308             Pair<List<String>, Map<ProcessorCapacityTickCountProperty, List<Long>>> instanceValuePair = ProcessorInformation
309                     .queryProcessorCapacityCounters();
310             instances = instanceValuePair.getA();
311             Map<ProcessorCapacityTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
312             systemList = valueMap.get(ProcessorCapacityTickCountProperty.PERCENTPRIVILEGEDTIME);
313             userList = valueMap.get(ProcessorCapacityTickCountProperty.PERCENTUSERTIME);
314             irqList = valueMap.get(ProcessorCapacityTickCountProperty.PERCENTINTERRUPTTIME);
315             softIrqList = valueMap.get(ProcessorCapacityTickCountProperty.PERCENTDPCTIME);
316             // % Processor Time is actually Idle time
317             idleList = valueMap.get(ProcessorCapacityTickCountProperty.PERCENTPROCESSORTIME);
318             baseList = valueMap.get(ProcessorCapacityTickCountProperty.TIMESTAMP_SYS100NS);
319             // Utility ticks, if configured
320             systemUtility = valueMap.get(ProcessorCapacityTickCountProperty.PERCENTPRIVILEGEDUTILITY);
321             processorUtility = valueMap.get(ProcessorCapacityTickCountProperty.PERCENTPROCESSORUTILITY);
322             processorUtilityBase = valueMap.get(ProcessorCapacityTickCountProperty.PERCENTPROCESSORUTILITY_BASE);
323             Map<ProcessorCapacityTickCountProperty, List<Long>> initialMap = ProcessorInformation
324                     .queryInitialProcessorCapacityCounters();
325             initSystemList = initialMap.get(ProcessorCapacityTickCountProperty.PERCENTPRIVILEGEDTIME);
326             initUserList = initialMap.get(ProcessorCapacityTickCountProperty.PERCENTUSERTIME);
327             initBase = initialMap.get(ProcessorCapacityTickCountProperty.TIMESTAMP_SYS100NS);
328             // Utility ticks, if configured
329             initSystemUtility = initialMap.get(ProcessorCapacityTickCountProperty.PERCENTPRIVILEGEDUTILITY);
330             initProcessorUtility = initialMap.get(ProcessorCapacityTickCountProperty.PERCENTPROCESSORUTILITY);
331             initProcessorUtilityBase = initialMap.get(ProcessorCapacityTickCountProperty.PERCENTPROCESSORUTILITY_BASE);
332         } else {
333             Pair<List<String>, Map<ProcessorTickCountProperty, List<Long>>> instanceValuePair = ProcessorInformation
334                     .queryProcessorCounters();
335             instances = instanceValuePair.getA();
336             Map<ProcessorTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
337             systemList = valueMap.get(ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME);
338             userList = valueMap.get(ProcessorTickCountProperty.PERCENTUSERTIME);
339             irqList = valueMap.get(ProcessorTickCountProperty.PERCENTINTERRUPTTIME);
340             softIrqList = valueMap.get(ProcessorTickCountProperty.PERCENTDPCTIME);
341             // % Processor Time is actually Idle time
342             idleList = valueMap.get(ProcessorTickCountProperty.PERCENTPROCESSORTIME);
343         }
344 
345         int ncpu = getLogicalProcessorCount();
346         long[][] ticks = new long[ncpu][TickType.values().length];
347         if (instances.isEmpty() || systemList == null || userList == null || irqList == null || softIrqList == null
348                 || idleList == null
349                 || (ProcessorInformation.USE_CPU_UTILITY && (baseList == null || systemUtility == null
350                         || processorUtility == null || processorUtilityBase == null || initSystemList == null
351                         || initUserList == null || initBase == null || initSystemUtility == null
352                         || initProcessorUtility == null || initProcessorUtilityBase == null))) {
353             return ticks;
354         }
355         for (int p = 0; p < instances.size(); p++) {
356             int cpu = instances.get(p).contains(",") ? numaNodeProcToLogicalProcMap.getOrDefault(instances.get(p), 0)
357                     : ParseUtil.parseIntOrDefault(instances.get(p), 0);
358             if (cpu >= ncpu) {
359                 continue;
360             }
361             ticks[cpu][TickType.SYSTEM.getIndex()] = systemList.get(cpu);
362             ticks[cpu][TickType.USER.getIndex()] = userList.get(cpu);
363             ticks[cpu][TickType.IRQ.getIndex()] = irqList.get(cpu);
364             ticks[cpu][TickType.SOFTIRQ.getIndex()] = softIrqList.get(cpu);
365             ticks[cpu][TickType.IDLE.getIndex()] = idleList.get(cpu);
366 
367             // If users want Task Manager output we have to do some math to get there
368             if (ProcessorInformation.USE_CPU_UTILITY) {
369                 // We have two new capacity numbers, processor (all but idle) and system
370                 // (included in processor). To further complicate matters, these are in percent
371                 // units so must be divided by 100.
372 
373                 // Calculate Utility CPU usage since init
374                 long deltaBase = processorUtilityBase.get(cpu) - initProcessorUtilityBase.get(cpu);
375 
376                 if (deltaBase > 0) {
377                     long deltaSys = systemUtility.get(cpu) - initSystemUtility.get(cpu);
378                     double sysPercent = deltaSys / (double) deltaBase; // subset of proc
379                     long deltaProc = processorUtility.get(cpu) - initProcessorUtility.get(cpu);
380                     double userPercent = deltaProc / (double) deltaBase - sysPercent;
381 
382                     // Apply to elapsed ticks
383                     long deltaT = baseList.get(cpu) - initBase.get(cpu);
384                     long newUser = initUserList.get(cpu) + Math.round(deltaT * userPercent / 100d);
385                     long newSystem = initSystemList.get(cpu) + Math.round(deltaT * sysPercent / 100d);
386 
387                     // Adjust user to new, saving the delta
388                     long delta = newUser - ticks[cpu][TickType.USER.getIndex()];
389                     ticks[cpu][TickType.USER.getIndex()] = newUser;
390                     // Do the same for system
391                     delta += newSystem - ticks[cpu][TickType.SYSTEM.getIndex()];
392                     ticks[cpu][TickType.SYSTEM.getIndex()] = newSystem;
393                     // Subtract delta from idle
394                     ticks[cpu][TickType.IDLE.getIndex()] -= delta;
395                 }
396             }
397 
398             // Decrement IRQ to avoid double counting in the total array
399             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
400                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
401 
402             // Raw value is cumulative 100NS-ticks
403             // Divide by 10_000 to get milliseconds
404             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10_000L;
405             ticks[cpu][TickType.USER.getIndex()] /= 10_000L;
406             ticks[cpu][TickType.IRQ.getIndex()] /= 10_000L;
407             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10_000L;
408             ticks[cpu][TickType.IDLE.getIndex()] /= 10_000L;
409         }
410         // Skipping nice and IOWait, they'll stay 0
411         return ticks;
412     }
